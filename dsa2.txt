#include <stdio.h>

int* read() {
    static int arr[5];    // static array (lives after function ends)

    printf("Enter 5 values:\n");
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }

    return arr;           // return pointer to array
}

void display(int arr[]) {
    printf("Array elements:\n");
    for (int i = 0; i < 5; i++) {
        printf("%d\n", arr[i]);
    }
}

int main() {
    int *arr;           // pointer to hold returned array

    arr = read();       // receive array from read()
    display(arr);       // pass array to display()

    return 0;
}
